[tex]u2 = 2(2) {}^{2} - 2 - 2 \\ u2 = 8 - 2 - 2 \\ u2 = 4[/tex]

U3 :

[tex]u3 = 2(3) {}^{2} - 3 - 2 \\ u3 = 18 - 3 - 2 \\ u3 = 13[/tex]

U4 :

[tex]u4 = 2(4) {}^{2} - 4 - 2 \\ u4 = 32 - 4 - 2 \\ u4 = 26[/tex]

b.

[tex]un = \frac{1}{2} n \: (n + 2) \\[/tex]

U1 :

[tex]u1 = \frac{1}{2} 1(1 + 2) \\ u1 = \frac{1}{2} \times 3 \\ u1 = 1.5[/tex]

U2 :

[tex]u2 = \frac{1}{2} 2(2 + 2) \\ u2 = \frac{1}{2} \times 2 \times 4 \\ u2 = \frac{1}{2} \times 8 \\ u2 = 4[/tex]

U3 :

[tex]u3 = \frac{1}{2} 3(3 + 2) \\ u3 = \frac{1}{2} \times 3 \times 5 \\ u3 = \frac{1}{2} \times 15 \\ u3 = 7.5[/tex]

U4 :

[tex]u4 = \frac{1}{2} 4(4 + 2) \\ u4 = \frac{1}{2} \times 4 \times 6 \\ u4 = \frac{1}{2} \times 24 \\ u4 = 12[/tex]
